# Import Required Library
import cv2
import numpy as np
from matplotlib import pyplot as plt
# Read the image
image = cv2.imread('path_to_your_image.jpg', 0)
# Apply edge detection (can use Canny or other edge detectors)
edges = cv2.Canny(image, 50, 150, apertureSize=3)
# Apply Hough Line Transform
lines = cv2.HoughLines(edges, 1, np.pi / 180, 200) # Adjust parameters as needed
# Draw detected lines on a copy of the original image
hough_image = np.copy(image)
if lines is not None:
for line in lines:
rho, theta = line[0]
a = np.cos(theta)
b = np.sin(theta)
x0 = a * rho
y0 = b * rho
x1 = int(x0 + 1000 * (-b)) # Set the line length as needed
y1 = int(y0 + 1000 * (a)) # Set the line length as needed
x2 = int(x0 - 1000 * (-b)) # Set the line length as needed
y2 = int(y0 - 1000 * (a)) # Set the line length as needed
cv2.line(hough_image, (x1, y1), (x2, y2), (0, 0, 255), 2)
# Display the Hough Transform image
plt.imshow(hough_image, cmap='gray')
plt.title('Hough Transform')
plt.axis('off')
plt.show()
